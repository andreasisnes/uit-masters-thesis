\begin{align}\label{eq:imbalance}
    imbalance = 
    \frac{
    \begin{bmatrix}
        p^a_1 \\
        \vdots \\
        p^a_n
    \end{bmatrix}
    \cdot
    \begin{bmatrix}
        v^a_1 \dots v^a_n
    \end{bmatrix}
    }{
    \begin{bmatrix}
        p^b_1 \\
        \vdots \\
        p^b_n
    \end{bmatrix}
    \cdot
    \begin{bmatrix}
        v^b_1 \dots v^b_n
    \end{bmatrix}
    }
    = \frac{\langle P_a, V_a \rangle}{\langle P_b, V_b \rangle}
\end{align}
